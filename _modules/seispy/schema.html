[docs]    def get_block(self):
        """
        .. todo::

           Document this method.
        """
        tokens = []
        char = self.next()
        if char == None:
            return None
        self.block = char
        while char != ';' or tokens:
            char = self.next()
            if char == None:
                self.block = None
                return None
            if tokens and char == tokens[-1]:
                tokens.pop()
            elif char in _lead_tokens:
                tokens += [_token_conjugates[char]]
            self.block += char
        self.block_cursor = 0
        return self.block



[docs]    def get_field(self):
        """
        .. todo::

           Document this method.
        """
        self.field = ''
        tokens = []
        while self.block_cursor < len(self.block):
            char = self.block[self.block_cursor]
            self.block_cursor += 1
            if tokens and char == tokens[-1]:
                tokens.pop()
            elif char in _lead_tokens:
                tokens += [_token_conjugates[char]]
            if not tokens and char == '\n':
                if self.field == '':
                    self.get_field()
                return self.field
            self.field += char
        self.field = None
        return None

        i = 0

[docs]class Schema:
    """
    Container class providing storage for schema definitions.

    This class also provides functionality to write Sphinx
    documentation pages for schema definitions.

    :keyword str path: path to schema definition
    :keyword str schema: schema name

    If ``path`` is provided, Schema instance will be initialized
    from definition found in ``path``. If ``path`` is not
    provided and ``schema`` is, Schema instance will be
    initialized from installed package definition of ``schema``.

    +------------------+---------------------+
    |Public Attributes |Private Attributes   |
    +==================+=====================+
    |:attr:`attributes`|:attr:`_token_parser`|
    +------------------+                     +
    |:attr:`relations` |                     |
    +------------------+---------------------+

    .. attribute:: Schema.attributes

       The ``attributes`` attribute is a :class:`tuple` of
       :class:`_Attribute` objects. These are the attributes (a.k.a.
       fields) used by the schema.

    .. attribute:: Schema.relations

       The ``relations`` attribute is a :class:`tuple`

    .. attribute:: _token_parser

       A :class:`_TokenParser` instance to parse schema definition.

    +-------------------------+-----------------------------------+
    |Public Methods           |Private Methods                    |
    +=========================+===================================+
    |:meth:`write_sphinx_docs`|:meth:`_parse_attribute`           |
    +-------------------------+-----------------------------------+
    |                         |:meth:`_parse_body`                |
    +                         +-----------------------------------+
    |                         |:meth:`_parse_header`              |
    +                         +-----------------------------------+
    |                         |:meth:`_parse_relation`            |
    +                         +-----------------------------------+
    |                         |:meth:`_write_schema_elements`     |
    +                         +-----------------------------------+
    |                         |:meth:`_write_schema_element_index`|
    +                         +-----------------------------------+
    |                         |:meth:`_write_schema_index`        |
    +-------------------------+-----------------------------------+

    .. automethod:: write_sphinx_docs

    .. automethod:: _parse_attribute

    .. automethod:: _parse_body

    .. automethod:: _parse_header

    .. automethod:: _parse_relation

    .. automethod:: _write_schema_elements

    .. automethod:: _write_schema_element_index

    .. automethod:: _write_schema_index

    .. versionadded:: 0.0alpha
    """
    def __init__(self, path=None, schema=seispy._DEFAULT_SCHEMA):
        self.attributes = {}
        self.relations = {}
        if path:
            self._token_parser = _TokenParser(path)
        else:
            self._token_parser = _TokenParser("%s/%s" % (seispy._DATA_DIR,
                                                        schema)
                                            )
        self._parse_header()
        self._parse_body()
        self.attributes = tuple(sorted([self.attributes[att] for att\
                in self.attributes],
                                       key=sort_key)
                               )
        self.relations = tuple(sorted([self.relations[rel] for rel in self.relations],
                                      key=sort_key)
                              )

    def __str__(self):
        s = ''
        for relation in self.relations:
            s += "{}\n".format(self.relations[relation])
        return s


[docs]    def _parse_body(self):
        """
        Parse body of schema definition.

        The body is considered all attribute and relation definitions.
        This method uses a :class:`_TokenParser` to read coherent
        blocks of data and calls :meth:`_parse_attribute` or
        :meth:`_parse_relation`, depending on the result returned
        by :meth:`_TokenParser.get_key`.
        """
        tp = self._token_parser
        while tp.get_block():
            tp.get_field()
            if tp.get_key() == 'Attribute':
                self._parse_attribute()
            elif tp.get_key() == 'Relation':
                self._parse_relation()
            else:
                raise Exception("Error code 3000: unrecognized specifier "\
                        "{:s}".format(tp.get_key()))



[docs]    def _parse_attribute(self):
        """
        Parse contents of an attribute definition.

        Assuming the block of data currently stored by
        :attr:`_token_parser` is an attribute definition, this method
        will parse the contents of the block and will add to
        :attr:`attributes` a new :class:`_Attribute` instance
        representing the block contents.
        """
        tp = self._token_parser
        kwargs = {}
        name = tp.get_value()
        kwargs['name'] = name
        while tp.get_field():
            key = tp.get_key()
            if key in _attribute_dtypes:
                kwargs['dtype'] = key.lower()
                kwargs['width'] = tp.get_value()
            elif key == 'Detail':
                lines = tp.get_value().strip('"').splitlines()
                kwargs['detail'] = '\n'.join([line.lstrip() for line in \
                        tp.get_value().strip('"').splitlines()])

            else:
                kwargs[key.lower()] = ' '.join(tp.get_value().strip('"').split())
        self.attributes[name] = _Attribute(**kwargs)



[docs]    def _parse_relation(self):
        """
        Parse contents of a relation definition.

        Assuming the block of data currently stored by
        :attr:`_token_parser` is an relation definition, this method
        will parse the contents of the block and will add to
        :attr:`relations` a new :class:`_Relation` instance
        representing the block contents.
        """
        tp = self._token_parser
        kwargs = {}
        name = tp.get_value()
        kwargs['name'] = name
        kwargs['transient'] = False
        while tp.get_field():
            key = tp.get_key()
            if key == 'Fields' or\
                    key == 'Primary' or\
                    key == 'Alternate' or\
                    key == 'Foreign' or\
                    key == 'Defines':
                if key == 'Fields':
                    key = 'fields'
                elif key == 'Primary':
                    key = 'primary_keys'
                elif key == 'Alternate':
                    key = 'alternate_keys'
                elif key == 'Foreign':
                    key = 'foreign_keys'
                elif key == 'Defines':
                    key = 'defines'
                kwargs[key] = []
                for attr in tp.get_value().split():
                    if '::' in attr:
                        attr1, attr2 = attr.split('::')
                        kwargs[key] += [(self.attributes[attr1],
                                         self.attributes[attr2])]
                    else:
                        kwargs[key] += [self.attributes[attr]]
                kwargs[key] = sorted(kwargs[key], key=sort_key)
            elif key == 'Transient':
                kwargs['transient'] = True
            else:
                kwargs[tp.get_key().lower()] = ' '.join(tp.get_value().strip('"').split())
        self.relations[name] = _Relation(**kwargs)



[docs]    def _parse_header(self):
        """
        Parse header contents of schema definition.

        Schema definitions begin with a header to provide the name,
        description and details about the schema. This method will
        parse the contents of the schema definition header, with the
        ability to handle *Schema*, *Description*, *Detail*, and
        *Timedate* keywords.
        """
        tp = self._token_parser
        tp.get_block()
        while tp.get_field():
            if tp.get_key() == 'Schema':
                self.schema = tp.get_value().strip('"')
            if tp.get_key() == 'Description':
                self.description= tp.get_value().strip('"')
            if tp.get_key() == 'Detail':
                self.detail = tp.get_value().strip('"')
            if tp.get_key() == 'Timedate':
                self.timedate = tp.get_value()



[docs]    def write_sphinx_docs(self, output_dir):
        """
        Generate Sphinx documentation for Schema instance.

        :argument str output_dir: path to target output directory with
                                  write access
        """
        if not isdir(output_dir):
            mkdir(output_dir)
        subdir = "%s/%s" % (output_dir, self.schema)
        if isdir(subdir):
            rmtree(subdir)
        mkdir(subdir)
        mkdir("%s/attributes" % subdir)
        mkdir("%s/relations" % subdir)
        attr_indices, rel_indices = [], []
        for attr in self.attributes:
            if attr.name[0].upper() not in attr_indices:
                attr_indices += [attr.name[0].upper()]
        for rel in self.relations:
            if rel.name[0].upper() not in rel_indices:
                rel_indices += [rel.name[0].upper()]
        self._write_schema_index(subdir,
                                attr_indices=attr_indices,
                                rel_indices=rel_indices)
        self._write_schema_element_index(subdir,
                                        elements='attributes',
                                        indices=attr_indices)
        self._write_schema_element_index(subdir,
                                        elements='relations',
                                        indices=rel_indices)
        for char in attr_indices:
            self._write_schema_element_index(subdir,
                                            key=char,
                                            elements='attributes')
        for char in rel_indices:
            self._write_schema_element_index(subdir,
                                            key=char,
                                            elements='relations')
        attr_indices = self._write_schema_elements(subdir, elements='attributes')
        rel_indices = self._write_schema_elements(subdir, elements='relations')



[docs]    def _write_schema_index(self, subdir, attr_indices=None, rel_indices=None):
        """
        Generate index page for schema instance.

        Index page consists of links to alphabetized indices of both
        schema attribute definitions and schema relation definitions,
        as well as details on the schema if available.

        :argument str subdir: The name of the sub-directory within the
                              target output directory to write schema
                              index page to.
        :keyword list attr_indices: a list of alphabetic indices that
                                    have a non-zero set of attributes
        :keyword list rel_indices: a list of alphabetic indices that
                                   have a non-zero set of relations
        """
        fout = open("%s/%s_index.rst" % (subdir, self.schema), 'w')
        fout.write(".. _%s_schema_index:\n\n" % self.schema)
        if hasattr(self, 'description'):
            title = "**%s** -- %s" % (self.schema, self.description)
        else:
            title = "**%s**" % self.schema
        fout.write("%s\n" % title + "=" * len(title) + "\n\n")
        fout.write(".. toctree::\n")
        fout.write("   :hidden:\n\n")
        fout.write("   attributes/attributes_index_all\n")
        fout.write("   relations/relations_index_all\n\n")
        if attr_indices:
            fout.write("Attributes\n----------\n\n")
            index = ""
            for char in attr_indices:
                index += ":ref:`%s <%s_attributes_index_%s>` | " %\
                        (char, self.schema, char)
            index += ":ref:`all <%s_attributes_index_all>`" % self.schema
            fout.write("%s\n\n" % index)
        if rel_indices:
            fout.write("Relations\n---------\n\n")
            index = ""
            for char in rel_indices:
                index += ":ref:`%s <%s_relations_index_%s>` | " %\
                        (char, self.schema, char)
            index += ":ref:`all <%s_relations_index_all>`" % self.schema
            fout.write("%s\n\n" % index)
        if hasattr(self, 'detail'):
            fout.write("Detail\n------\n\n")
            fout.write("%s\n" % self.detail)
        fout.close()



[docs]    def _write_schema_element_index(self,
                                   subdir,
                                   key='all',
                                   elements=None,
                                   indices=None):
        """
        Generate index page for indexed schema elements.

        A schema element is an attribute or a relation. This method
        will generate an index page for schema elements grouped by
        first letter. (Eg.  an index page for all attributes falling
        under the index 'A' - *algorithm*, *amp*, *anet*...)

        :argument str subdir: the name of the sub-directory within the
                              target output directory to write grouped
                              schema element index page to
        :keyword str key: the key (character) to generate index page
                          for
        :keyword str elements: the schema element (attributes,
                               relations) to generate index page for
        :keyword list indices: a list of indices with a non-zero set of
                               indices required to generate the index
                               page for key *all*
        """
        if key == 'all':
            objs = getattr(self, elements)
        else:
            objs = [obj for obj in getattr(self, elements) if\
                    obj.name[0].upper() == key]
        if len(objs) == 0:
            return
        fout = open("%s/%s/%s_index_%s.rst" % (subdir, elements, elements, key), 'w')
        fout.write(".. _%s_%s_index_%s:\n\n" % (self.schema, elements, key))
        title = "%s%s Index -- **%s**" % (elements[0].upper(), elements[1:], key)
        fout.write("%s\n" % title + "=" * len(title) + "\n\n")
        fout.write(draw_table(objs, schema=self.schema) + "\n")
        fout.write(".. toctree::\n")
        fout.write("   :hidden:\n\n")
        if key == 'all':
            for char in indices:
                fout.write("   %s_index_%s\n" % (elements, char))
        else:
            for obj in objs:
                fout.write("   %s\n" % obj.name)
        fout.close()

[docs]def draw_table(iterator,
               header=None,
               schema="NULLSCHEMA",
               ncol=6):
    """
    .. todo::

       Document this function.

    """
    if len(iterator) < ncol:
        ncol = len(iterator)
    cell_entries = {}
    for element in iterator:
        if isinstance(element, tuple):
            key = "%s::%s" % (element[0].name, element[1].name)
            entry = ":ref:`%s <%s-%s_attributes>`::"\
                    ":ref:`%s <%s-%s_attributes>`" %\
                    (element[0].name,
                     schema,
                     element[0].name,
                     element[1].name,
                     schema,
                     element[1].name)
        else:
            if isinstance(element, _Attribute):
                suffix = "_attributes"
            elif isinstance(element, _Relation):
                suffix = "_relations"
            key = element.name
            #entry = ":doc:`%s </source/schemas/%s/%s/%s.rst>`" %\
            #        (element.name, schema, subdir, element.name)
            entry = ":ref:`%s <%s-%s%s>`" %\
                    (element.name, schema, element.name, suffix)
        cell_entries[key] = entry
    colwidth = max([len(cell_entries[key]) for key in cell_entries])
    tblwidth = ncol * (colwidth + 1) + 1
    if header:
        s = "+" + "-" * (tblwidth - 2) + "+\n"
        s += "|%s" % header + " " * (tblwidth - len("|%s" % header) - 1) + "|\n"
        s += ("+" + "=" * colwidth) * ncol + "+\n"
    else:
        s = ("+" + "-" * colwidth) * ncol + "+\n"
    counter = 1
    for key in sorted(cell_entries):
        entry = cell_entries[key]
        s += "|%s" % entry + " " * (colwidth - len(entry))
        if counter % ncol == 0:
            s += "|\n"
            if counter != len(cell_entries):
                s += ("+" + "-" * colwidth) * ncol + "+" + "\n"
        counter += 1
    if len(cell_entries) % ncol != 0:
        for i in range(ncol - (len(cell_entries) % ncol)):
            s += "|" + " " * colwidth
        s += "|\n"
    s += ("+" + "-" * colwidth) * ncol + "+\n"
    return s
